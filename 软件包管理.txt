rpm命令管理：
	rpm -ivh 包全名 （RPM安装）
		-i：安装
		-v 显示详细信息
		-h 显示进度
		--nodeps 不检测依赖性
	rpm -Uvh 包全名 （RPM包升级）
	rpm -e 包名 （卸载）
	
	查询是否安装：rpm -q 包名
		        rpm -qa （查询所有包）
	
	查询包详细信息：rpm -qi 包名
		-i 查询软件信息
		-p （后跟包全名）查询未安装包信息

	rpm -ql 包名 （查询包的位置）
		-l 列表
		-p （后跟包全名）未安装包信息

	rpm -qf 文件路径 （查询系统文件所属包）
		-f 查询系统文件所属软件包

	rpm -qR 包名 （查看依赖性）
		-R 软件包依赖性
		-p （后跟包全名）未安装包依赖性
	
	RPM包校验：
		rpm -V 包名 （检查软件包是否被修改）
		修改内容（略）
	
	RPM包中文件提取：
		rpm2cpio 包全名 | \		//"\"代表换行
			> cpio -idev .文件绝对路径	//"." 提取指定文件放在当前目录下

yum在线管理：
（可用光盘作yum源，Redhat付费，centos 免费）
	IP地址配置
	网络yum源

	yum list (获取服务器可用软件包列表)

	yum search 关键字 （搜索服务器上含关键字的包）
	
	yum -y install 包名 （-y自动"是"安装包）
	             update 包名 （升级；缺省包名全升级（不可））
		
	yum -y remove 包名 （包名不可缺省；依赖包同样卸载）
		可能会卸载系统的依赖包
	
	yum grouplist （列出所有软件组）

	yum groupinstall 软件组名（安装指定软件组）

	yum groupremove 软件组名 （卸载指定软件组）

	光盘yum源搭建
